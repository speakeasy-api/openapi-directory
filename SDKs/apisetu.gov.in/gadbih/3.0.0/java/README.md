## SDK Example Usage

```java
package hello.world;

import openapisdk.SDK;
import openapisdk.models.shared.Security;

public class Application {
    public static void main(String[] args) {
        try {
            SDK.Builder builder = SDK.builder();

            SDK sdk = builder.build();

            CtcerRequest req = new CtcerRequest() {{
                security = new CtcerSecurity() {{
                    apiKey = new SchemeApiKey() {{
                        apiKey = "YOUR_API_KEY_HERE";
                    }};
                    clientId = new SchemeClientId() {{
                        apiKey = "YOUR_API_KEY_HERE";
                    }};
                }};
                request = new CtcerRequestBody() {{
                    certificateParameters = new CtcerRequestBodyCertificateParameters() {{
                        refNo = "corporis";
                        tokenNo = "accusantium";
                    }};
                    consentArtifact = "eaque";
                    format = "pdf";
                    txnId = "quidem";
                }};
            }};

            CtcerResponse res = sdk.apIs.ctcer(req);

            if (res.statusCode == 200) {
                // handle response
            }
        } catch (Exception e) {
            // handle exception
        }
```
