            PostEventsV3SendRequest req = new PostEventsV3SendRequest() {{
                security = new PostEventsV3SendSecurity() {{
                    hapikey = new SchemeHapikey() {{
                        apiKey = "YOUR_API_KEY_HERE";
                    }};
                }};
                request = new BehavioralEventHttpCompletionRequest() {{

                    }};

                }};
            }};
