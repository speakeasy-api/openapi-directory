import { BadWordFilterRequest, BadWordFilterResponse } from "openapi/src/sdk/models/operations";
import { AxiosError } from "axios";

const sdk = new SDK(withSecurity(
  security: {
    apiKey: {
      apiKey: "YOUR_API_KEY_HERE",
    },
    userId: {
      apiKey: "YOUR_API_KEY_HERE",
    },
  }

    censorCharacter: "voluptas",
