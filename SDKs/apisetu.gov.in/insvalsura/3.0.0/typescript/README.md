  },
  request: {
    certificateParameters: {

      personalno: "voluptas",
      unit: "culpa",
    },
    consentArtifact: "expedita",
    format: "pdf",
    txnId: "dolor",
  },
};

sdk.apIs.skcer(req).then((res: SkcerResponse | AxiosError) => {
   // handle response
});
