  },
  request: {
    certificateParameters: {
      custID: "sit",
      dob: "voluptas",
      finYr: "culpa",
    },
    consentArtifact: "expedita",
    format: "pdf",
    txnId: "dolor",
  },
};

sdk.apIs.tdcer(req).then((res: TdcerResponse | AxiosError) => {
   // handle response
});
